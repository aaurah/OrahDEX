🧮 3. BONDING CURVE MATH

We use a linear bonding curve (Zora style):

price = basePrice + slope \cdot supply


Where:

• basePrice = starting price
• slope = controls depth
• supply = circulating supply


To simulate $9k liquidity, choose:

basePrice = 0.0001 ETH
slope = 0.0000001 ETH


You can tune this per Keeper asset.

✅ Solidity Contract: Virtual Liquidity AMM

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

contract OrahDexVirtualAMM {

    // Token supply
    uint256 public totalSupply;

    // Bonding curve parameters
    uint256 public basePrice;   // e.g. 0.0001 ETH
    uint256 public slope;       // e.g. 0.0000001 ETH

    // Treasury balance (ETH)
    uint256 public treasury;

    // Mapping of user balances
    mapping(address => uint256) public balanceOf;

    // Events
    event Buy(address indexed user, uint256 amount, uint256 cost);
    event Sell(address indexed user, uint256 amount, uint256 payout);
    event PriceUpdated(uint256 newPrice);
    event TreasuryFunded(uint256 amount);

    constructor(uint256 _basePrice, uint256 _slope) {
        basePrice = _basePrice;
        slope = _slope;
    }

    // Current price = basePrice + slope * supply
    function getPrice() public view returns (uint256) {
        return basePrice + (slope * totalSupply);
    }

    // Cost to buy N tokens
    function getBuyCost(uint256 amount) public view returns (uint256) {
        uint256 startPrice = getPrice();
        uint256 endPrice = basePrice + slope * (totalSupply + amount);
        return (startPrice + endPrice) * amount / 2;
    }

    // Payout for selling N tokens
    function getSellPayout(uint256 amount) public view returns (uint256) {
        uint256 startPrice = getPrice();
        uint256 endPrice = basePrice + slope * (totalSupply - amount);
        return (startPrice + endPrice) * amount / 2;
    }

    // BUY
    function buy(uint256 amount) external payable {
        uint256 cost = getBuyCost(amount);
        require(msg.value >= cost, "Insufficient ETH");

        // Mint tokens
        balanceOf[msg.sender] += amount;
        totalSupply += amount;

        // Add ETH to treasury
        treasury += cost;

        emit Buy(msg.sender, amount, cost);
        emit PriceUpdated(getPrice());
    }

    // SELL
    function sell(uint256 amount) external {
        require(balanceOf[msg.sender] >= amount, "Not enough tokens");

        uint256 payout = getSellPayout(amount);
        require(treasury >= payout, "Treasury empty");

        // Burn tokens
        balanceOf[msg.sender] -= amount;
        totalSupply -= amount;

        // Pay user
        treasury -= payout;
        payable(msg.sender).transfer(payout);

        emit Sell(msg.sender, amount, payout);
        emit PriceUpdated(getPrice());
    }

    // Fund treasury (owner or Keeper)
    function fundTreasury() external payable {
        treasury += msg.value;
        emit TreasuryFunded(msg.value);
    }
}

Quote Engine

export function getBuyQuote(amount, supply, basePrice, slope) {
  const start = basePrice + slope * supply;
  const end = basePrice + slope * (supply + amount);
  return ((start + end) / 2) * amount;
}

export function getSellQuote(amount, supply, basePrice, slope) {
  const start = basePrice + slope * supply;
  const end = basePrice + slope * (supply - amount);
  return ((start + end) / 2) * amount;
}


---

Slippage Calculation

export function getSlippage(amount, supply, basePrice, slope) {
  const priceBefore = basePrice + slope * supply;
  const priceAfter = basePrice + slope * (supply + amount);
  return (priceAfter - priceBefore) / priceBefore;
}
