⭐ 1. Liquidity Check Logic (Core Rule)

Definition of liquidity

You consider liquidity available when:

• There is enough bid/ask depth in your orderbook
• The order can be filled without exceeding slippage
• The orderbook has at least one matching level


Core logic

function hasInternalLiquidity(orderbook, amountIn, side) {
    if (!orderbook || orderbook.length === 0) return false;

    let remaining = amountIn;
    let depth = side === 'buy' ? orderbook.asks : orderbook.bids;

    for (const level of depth) {
        if (remaining <= 0) return true;
        remaining -= level.amount;
    }

    return remaining <= 0;
}


This checks if your orderbook can fill the trade.

⭐ 2. Hybrid Routing Logic

If liquidity exists → internal execution

If liquidity does not exist → LetsExchange execution

async function routeSwap(pair, amountIn, side) {
    const orderbook = await getOrderbook(pair);

    const internal = hasInternalLiquidity(orderbook, amountIn, side);

    if (internal) {
        return executeInternalSwap(pair, amountIn, side);
    } else {
        return executeLetsExchangeSwap(pair, amountIn, side);
    }
}


This is the brain of your hybrid engine.

⭐ 4. LetsExchange Execution (Fallback)

A. Get Quote

async function getLEQuote(from, to, amount) {
    const res = await fetch(`https://api.letsexchange.io/api/v1/info/estimate?from=${from}&to=${to}&amount=${amount}`);
    return res.json();
}


B. Create Swap

async function createLESwap(from, to, amount, address) {
    const res = await fetch(`https://api.letsexchange.io/api/v1/transaction`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            fromCurrency: from,
            toCurrency: to,
            amount,
            address
        })
    });

    return res.json();
}


C. Execute Swap

async function executeLetsExchangeSwap(pair, amountIn, side) {
    const [from, to] = pair.split('/');

    const quote = await getLEQuote(from, to, amountIn);

    const tx = await createLESwap(from, to, amountIn, userWalletAddress);

    return {
        source: 'letsexchange',
        quote,
        tx
    };
}

app.post('/swap/execute', async (req, res) => {
    const { pair, amountIn, side } = req.body;

        const result = await routeSwap(pair, amountIn, side);
