1. Canonical flow for adding liquidity into a vAMM

For a vAMM-style DEX (no real reserves in the pool, only virtual balances):

1. User side (frontend/wallet):• Approve tokens to the router/pool:• token0.approve(router, amount0Desired)
• token1.approve(router, amount1Desired)

• Call addLiquidity(...) on router (or pool) with:• token0, token1
• amount0Desired, amount1Desired
• amount0Min, amount1Min
• to (LP receiver)
• deadline


2. Router/pool side:• Pull tokens via transferFrom
• Compute actual amounts used (respecting ratio + slippage)
• Update virtual reserves (for vAMM pricing)
• Mint LP shares proportional to contribution
• Emit events

2. Minimal vAMM + liquidity skeleton (Solidity-style)

This is a clean base you can align Orah with:

pragma solidity ^0.8.20;

interface IERC20 {
    function transferFrom(address from, address to, uint256 value) external returns (bool);
    function transfer(address to, uint256 value) external returns (bool);
    function balanceOf(address who) external view returns (uint256);
    function decimals() external view returns (uint8);
}

contract OrahDEXVAMMPool {
    address public token0;
    address public token1;

    uint256 public vReserve0; // virtual reserve for pricing
    uint256 public vReserve1;

    uint256 public totalSupply;          // total LP shares
    mapping(address => uint256) public balanceOf;

    event Mint(address indexed provider, uint256 amount0, uint256 amount1, uint256 liquidity);

    constructor(address _token0, address _token1) {
        token0 = _token0;
        token1 = _token1;
    }

    // --- core view ---

    function getVirtualReserves() public view returns (uint256, uint256) {
        return (vReserve0, vReserve1);
    }

    // x * y = k style invariant on virtual reserves
    function getPrice() public view returns (uint256 price0In1) {
        require(vReserve0 > 0 && vReserve1 > 0, "NO_LIQ");
        // price of token0 in terms of token1
        price0In1 = (vReserve1 * 1e18) / vReserve0;
    }

    // --- liquidity logic ---

    function _mint(address to, uint256 liquidity) internal {
        totalSupply += liquidity;
        balanceOf[to] += liquidity;
    }

    function addLiquidity(
        uint256 amount0Desired,
        uint256 amount1Desired,
        uint256 amount0Min,
        uint256 amount1Min,
        address to
    ) external returns (uint256 amount0, uint256 amount1, uint256 liquidity) {
        require(to != address(0), "ZERO_TO");

        (uint256 _v0, uint256 _v1) = (vReserve0, vReserve1);

        if (_v0 == 0 && _v1 == 0) {
            // first LP sets the virtual curve
            amount0 = amount0Desired;
            amount1 = amount1Desired;
            liquidity = _sqrt(amount0 * amount1);
        } else {
            // keep ratio consistent with current virtual reserves
            uint256 amount1Optimal = (amount0Desired * _v1) / _v0;
            if (amount1Optimal <= amount1Desired) {
                require(amount1Optimal >= amount1Min, "INSUFFICIENT_AMOUNT1");
                amount0 = amount0Desired;
                amount1 = amount1Optimal;
            } else {
                uint256 amount0Optimal = (amount1Desired * _v0) / _v1;
                require(amount0Optimal >= amount0Min, "INSUFFICIENT_AMOUNT0");
                amount0 = amount0Optimal;
                amount1 = amount1Desired;
            }

            liquidity = _min(
                (amount0 * totalSupply) / _v0,
                (amount1 * totalSupply) / _v1
            );
        }

        require(liquidity > 0, "INSUFFICIENT_LIQ_MINTED");

        // pull tokens from LP
        require(IERC20(token0).transferFrom(msg.sender, address(this), amount0), "T0_TRANSFER_FAIL");
        require(IERC20(token1).transferFrom(msg.sender, address(this), amount1), "T1_TRANSFER_FAIL");

        // update virtual reserves (vAMM)
        vReserve0 = _v0 + amount0;
        vReserve1 = _v1 + amount1;

        _mint(to, liquidity);

        emit Mint(to, amount0, amount1, liquidity);
    }

    // --- helpers ---

    function _min(uint256 x, uint256 y) internal pure returns (uint256) {
        return x < y ? x : y;
    }

    function _sqrt(uint256 y) internal pure returns (uint256 z) {
        if (y > 3) {
            z = y;
            uint256 x = y / 2 + 1;
            while (x < z) {
                z = x;
                x = (y / x + x) / 2;
            }
        } else if (y != 0) {
            z = 1;
        }
    }
}


You can adapt:

• Pure vAMM: you might not actually hold real tokens in the pool—then you’d skip transferFrom and only track virtual balances, while collateral sits elsewhere.
• Perpetuals-style vAMM: same invariant, but LP shares might map to funding fees, etc.

3. Usual reasons “add liquidity” fails

When you say “it’s giving error”, 90% of the time it’s one of these:

1. Missing approvals / wrong spender• Check: wallet approved router or pool address (not token or some other contract).
• If you call pool directly, spender must be the pool.

2. Slippage constraints too tight• amount0Min / amount1Min > what the ratio math produces → INSUFFICIENT_AMOUNT0/1-style revert.

3. Zero or tiny liquidity• First LP: if amount0Desired * amount1Desired is too small, _sqrt → 0 → INSUFFICIENT_LIQ_MINTED.

4. Token decimals / scaling mismatch• If one token is 6 decimals and you treat it as 18, ratios explode and revert.

5. Deadline expired• If you use a router pattern with deadline, any timestamp in the past reverts.

6. Wrong pair / wrong pool• Frontend might be pointing to a pool with different token0/token1 ordering than you expect.
